/* 新增角色相關的表 */
    CREATE TABLE functionStructure(
        `funcCode` VARCHAR(255) NOT NULL   COMMENT '功能代碼' ,
        `createTime` DATETIME    COMMENT '創建時間' ,
        `lastModify` DATETIME    COMMENT '最近修改' ,
        PRIMARY KEY (funcCode)
    )  COMMENT = '功能結構;';

    CREATE TABLE itemStructure(
        `itemCode` VARCHAR(255) NOT NULL   COMMENT '操作代碼' ,
        `createTime` DATETIME    COMMENT '創建時間' ,
        `lastModify` DATETIME    COMMENT '最近修改' ,
        PRIMARY KEY (itemCode)
    )  COMMENT = '操作代碼;';

    CREATE TABLE role(
        `companyId` INT NOT NULL   COMMENT '公司id' ,
        `roleId` INT NOT NULL AUTO_INCREMENT unique  COMMENT '角色id' ,
        `funcCode` VARCHAR(255) NOT NULL   COMMENT '功能代碼' ,
        `itemCode` VARCHAR(255) NOT NULL   COMMENT '操作代碼(edit, delete...)' ,
        `createTime` DATETIME    COMMENT '創建時間' ,
        `lastModify` DATETIME    COMMENT '最近修改' ,
        PRIMARY KEY (companyId,roleId,funcCode,itemCode)
    )  COMMENT = '角色表;';
